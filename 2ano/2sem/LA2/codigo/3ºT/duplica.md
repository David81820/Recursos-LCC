```Python

'''
Optimize a seguinte função.
'''


def duplica(lista, dic={}):
    if len(lista) == 0:
        return 0
    if len(lista) == 1:
        return 2 * lista[0]

    if tuple(lista) in dic:
        return dic[tuple(lista)]

    a = lista[0] + duplica(lista[1:], dic)
    b = 2 * lista[0] + lista[1] + duplica(lista[2:], dic)
    res = max(a, b)
    dic[tuple(lista)] = res
    return res

```
